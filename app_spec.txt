<project_specification>
  <project_name>Constituția României</project_name>

  <overview>
    Platformă web interactivă și modernă pentru explorarea Constituției României prin toate versiunile sale istorice (1952, 1986, 1991, 2003). Oferă navigare articol cu articol, vizualizare graph a structurii constituționale, comparare side-by-side între versiuni cu diff viewer modern, căutare instantanee cross-versiune, și feedback anonim moderat prin AI. Proiectul este complet în limba română, cu design bold civic-modern, fully responsive, dark/light mode.
  </overview>

  <language>ro</language>

  <technology_stack>

  <database_schema>
    <tables>
      <constitution_versions>
        - id (serial, PK)
        - year (integer, unique) — 1952, 1986, 1991, 2003
        - name (text) — "Constituția din 1952", etc.
        - description (text) — context istoric
        - source_file (text) — calea fișierului markdown sursă
        - total_articles (integer)
        - created_at (timestamp)
      </constitution_versions>

      <structural_units>
        - id (serial, PK)
        - version_id (FK → constitution_versions.id)
        - type (enum: 'titlu', 'capitol', 'sectiune')
        - number (integer) — numărul titlului/capitolului/secțiunii
        - name (text) — denumirea
        - parent_id (FK → structural_units.id, nullable) — pentru ierarhie (capitol → titlu)
        - order_index (integer) — ordinea în document
        - slug (text) — pentru URL-uri
      </structural_units>

      <articles>
        - id (serial, PK)
        - version_id (FK → constitution_versions.id)
        - structural_unit_id (FK → structural_units.id)
        - number (integer) — numărul articolului
        - title (text, nullable) — titlul articolului (doar 1991/2003)
        - content (text) — conținutul complet markdown
        - content_tiptap (jsonb) — conținut TipTap JSON
        - order_index (integer) — ordinea globală în document
        - slug (text) — pentru deep linking
        - agree_count (integer, default 0)
        - disagree_count (integer, default 0)
      </articles>

      <article_references>
        - id (serial, PK)
        - source_article_id (FK → articles.id)
        - target_article_id (FK → articles.id)
        - reference_text (text) — textul din sursă care face referirea
      </article_references>

      <comments>
        - id (serial, PK)
        - article_id (FK → articles.id)
        - content (text) — textul comentariului
        - selected_text (text, nullable) — textul selectat pentru inline comment
        - status (enum: 'pending', 'approved', 'rejected')
        - rejection_reason (text, nullable) — motivul respingerii de AI
        - fingerprint_hash (text) — hash browser fingerprint
        - ip_hash (text) — hash IP pentru rate limiting
        - created_at (timestamp)
      </comments>

      <votes>
        - id (serial, PK)
        - article_id (FK → articles.id)
        - vote_type (enum: 'agree', 'disagree')
        - fingerprint_hash (text) — prevenire voturi multiple
        - created_at (timestamp)
        - UNIQUE constraint pe (article_id, fingerprint_hash)
      </votes>
    </tables>
  </database_schema>
